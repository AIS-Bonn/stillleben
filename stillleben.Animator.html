Typical usage

import stillleben as sl

# First pose
p1 = torch.eye(4)

# Translated second pose
p2 = p1.clone()
p2[:3, 3] = torch.tensor([0.0, 1.0, 0.0])

# Animate for 100 ticks
anim = sl.Animator([p1, p2], 100)
for pose in anim:
    print(pose)
